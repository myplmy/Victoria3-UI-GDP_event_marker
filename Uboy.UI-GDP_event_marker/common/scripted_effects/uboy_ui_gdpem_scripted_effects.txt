# ==================================================================================================================== #

uboy_ui_gdpem_calculate_quarter_index = {
    set_local_variable = {
        name = uboy_ui_gdpem_years_elapsed
        value = year
    }
    change_local_variable = {
        name = uboy_ui_gdpem_years_elapsed
        subtract = global_var:uboy_ui_gdpem_game_start_year
    }
    
    # 현재 분기 계산 (0~3)
    set_local_variable = {
        name = uboy_ui_gdpem_current_quarter
        value = 0
    }
    if = {
        limit = {
            OR = {
                month = 4
                month = 5
                month = 6
            }
        }
        set_local_variable = { name = uboy_ui_gdpem_current_quarter value = 1 }
    }
    else_if = {
        limit = {
            OR = {
                month = 7
                month = 8
                month = 9
            }
        }
        set_local_variable = { name = uboy_ui_gdpem_current_quarter value = 2 }
    }
    else_if = {
        limit = {
            OR = {
                month = 10
                month = 11
                month = 12
            }
        }
        set_local_variable = { name = uboy_ui_gdpem_current_quarter value = 3 }
    }
    
    # 분기 인덱스 = 년수 * 4 + 현재 분기
    set_local_variable = {
        name = uboy_ui_gdpem_quarter_index
        value = local_var:uboy_ui_gdpem_years_elapsed
    }
    # multiply 대신 반복 add 사용
    change_local_variable = {
        name = uboy_ui_gdpem_quarter_index
        add = local_var:uboy_ui_gdpem_years_elapsed
    }
    change_local_variable = {
        name = uboy_ui_gdpem_quarter_index
        add = local_var:uboy_ui_gdpem_years_elapsed
    }
    change_local_variable = {
        name = uboy_ui_gdpem_quarter_index
        add = local_var:uboy_ui_gdpem_years_elapsed
    }
    # 위 3번 add = 원래 값 * 4
    
    change_local_variable = {
        name = uboy_ui_gdpem_quarter_index
        add = local_var:uboy_ui_gdpem_current_quarter
    }
}

# ==================================================================================================================== #
# Territory Change Detection
# ==================================================================================================================== #

uboy_ui_gdpem_check_territory_gained = {
    if = {
        limit = {
            has_variable = uboy_ui_gdpem_last_state_count
            num_states > var:uboy_ui_gdpem_last_state_count
        }
        uboy_ui_gdpem_record_event = { EVENT_TYPE = 4 }
    }
    
    set_variable = {
        name = uboy_ui_gdpem_last_state_count
        value = num_states
    }
}

uboy_ui_gdpem_check_territory_lost = {
    if = {
        limit = {
            has_variable = uboy_ui_gdpem_last_state_count
            num_states < var:uboy_ui_gdpem_last_state_count
        }
        uboy_ui_gdpem_record_event = { EVENT_TYPE = 5 }
    }
    
    set_variable = {
        name = uboy_ui_gdpem_last_state_count
        value = num_states
    }
}
