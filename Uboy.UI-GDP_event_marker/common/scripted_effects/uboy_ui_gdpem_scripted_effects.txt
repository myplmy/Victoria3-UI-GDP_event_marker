# ==================== 분기 계산 ====================

event_marker_calculate_quarter_index = {
    # global_variable로 현재 년도 저장
    set_global_variable = {
        name = event_marker_current_year
        value = year
    }
    
    # 경과 년수 계산
    set_local_variable = {
        name = elapsed_years
        value = {
            add = global_var:event_marker_current_year
            subtract = global_var:event_marker_game_start_year
        }
    }
    
    # 현재 분기 계산 (0~3)
    # 1~3월 = 0, 4~6월 = 1, 7~9월 = 2, 10~12월 = 3
    set_local_variable = {
        name = current_quarter_in_year
        value = 0
    }
    
    if = {
        limit = { month >= 1 month <= 3 }
        set_local_variable = { name = current_quarter_in_year value = 0 }
    }
    else_if = {
        limit = { month >= 4 month <= 6 }
        set_local_variable = { name = current_quarter_in_year value = 1 }
    }
    else_if = {
        limit = { month >= 7 month <= 9 }
        set_local_variable = { name = current_quarter_in_year value = 2 }
    }
    else_if = {
        limit = { month >= 10 month <= 12 }
        set_local_variable = { name = current_quarter_in_year value = 3 }
    }
    
    # 전체 분기 인덱스 = 경과년수 * 4 + 분기
    set_local_variable = {
        name = quarter_index

            add = local_var:elapsed_years
            multiply = 4
            add = local_var:current_quarter_in_year
        }
    }

    set_variable = {
        name = event_marker_current_quarter
        value = local_var:quarter_index
    }
}
